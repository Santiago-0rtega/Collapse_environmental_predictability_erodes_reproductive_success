S10. Sensitivity Analysis: Bloom Threshold Definition

This section details a sensitivity analysis to test the robustness of the phenological mismatch model to the definition of the bloom start date. The main analysis uses a 5% increase over the seasonal median chlorophyll-a as the threshold to define the bloom (X5MISMATCH). Here, we refit the best-supported Gaussian model for phenological mismatch using two alternative, more sensitive thresholds: a 15% increase (X15MISMATCH) and a 25% increase (X25MISMATCH).

Model fitting

A specialized workflow function, run_mismatch_sensitivity(), is defined to automate the process. This function takes the dataset and the specific mismatch variable to be used as the response, fits the top-ranked Gaussian model (age_slope_sigma_linear), and saves the results.

# --- Load Libraries ---
if (!require("pacman")) install.packages("pacman")
pacman::p_load(
    brms, dplyr, here, future, purrr, tibble
)
# --- Parallel Processing Setup ---
options(mc.cores = parallel::detectCores(), brms.backend = "cmdstanr")
plan(multisession, workers = 4)

# --- Define Fixed Effects Structure for the Best Gaussian Model ---
FIXED_EFFECTS_GAUSSIAN <- "AGE + I(AGE^2) + TIME + AFR + LONGEVITY"

# --- Workflow Function for Mismatch Sensitivity ---
run_mismatch_sensitivity <- function(data, response_var, sex) {
  
  analysis_name <- paste0("sensitivity_mismatches")
  message(paste0("\n###--- Starting ", toupper(sex), " Analysis for: '", response_var, "' ---###"))

  output_dir <- here::here("Rdata", analysis_name)
  dir.create(output_dir, recursive = TRUE, showWarnings = FALSE)
  
  # Define the single best model structure
  model_spec <- tribble(
      ~name,                      ~mean_random,                  ~aux_formula,
      "age_slope_sigma_linear",   "(1+AGE|q|RING) + (1|YEAR)",   ~ 1 + AGE + TIME + (1|q|RING),
  )
  
  model_name <- model_spec$name
  model_path <- file.path(output_dir, paste0(response_var, "_", sex, "_fit_", model_name, ".rds"))

  if (file.exists(model_path)) {
    message(paste0("Skipping '", model_name, "': Model file already exists."))
    return(invisible(NULL))
  }

  message(paste0("\n--- Fitting model: ", model_name, " ---"))
  
  formula_str <- paste(response_var, "~", FIXED_EFFECTS_GAUSSIAN, "+", model_spec$mean_random)
  brm_formula <- bf(formula_str, sigma = model_spec$aux_formula, nl = FALSE)
  
  model_priors <- get_prior(brm_formula, data = data, family = gaussian())

  fit <- tryCatch({
    brm(
      formula = brm_formula, data = data, prior = model_priors, family = gaussian(),
      iter = 6000, warmup = 2000, chains = 4, seed = 123, init = 0,
      control = list(adapt_delta = 0.99, max_treedepth = 15)
    )
  }, error = function(e) { message(paste0("ERROR fitting '", model_name, "': ", e$message)); NULL })

  if (!is.null(fit)) {
    saveRDS(fit, model_path)
  }
}

title: "S10. Sensitivity Analysis: Bloom Threshold Definition"
---

This section details a sensitivity analysis to test the robustness of the phenological mismatch model to the definition of the bloom start date. The main analysis uses a 5% increase over the seasonal median chlorophyll-a as the threshold to define the bloom (`X5MISMATCH`). Here, we refit the best-supported Gaussian model for phenological mismatch using two alternative, more sensitive thresholds: a 15% increase (`X15MISMATCH`) and a 25% increase (`X25MISMATCH`).

# Model fitting

A specialized workflow function, run_mismatch_sensitivity(), is defined to automate the process. This function takes the dataset and the specific mismatch variable to be used as the response, fits the top-ranked Gaussian model (age_slope_sigma_linear), and saves the results.

```{r}
# --- Load Libraries ---
if (!require("pacman")) install.packages("pacman")
pacman::p_load(
    brms, dplyr, here, future, purrr, tibble
)
```

```{r}
#| label: setup-mismatch-sensitivity
#| eval: false



# --- Parallel Processing Setup ---
options(mc.cores = parallel::detectCores(), brms.backend = "cmdstanr")
plan(multisession, workers = 4)

# --- Define Fixed Effects Structure for the Best Gaussian Model ---
FIXED_EFFECTS_GAUSSIAN <- "AGE + I(AGE^2) + TIME + AFR + LONGEVITY"

# --- Workflow Function for Mismatch Sensitivity ---
run_mismatch_sensitivity <- function(data, response_var, sex) {
  
  analysis_name <- paste0("sensitivity_mismatches")
  message(paste0("\n###--- Starting ", toupper(sex), " Analysis for: '", response_var, "' ---###"))

  output_dir <- here::here("Rdata", analysis_name)
  dir.create(output_dir, recursive = TRUE, showWarnings = FALSE)
  
  # Define the single best model structure
  model_spec <- tribble(
      ~name,                      ~mean_random,                  ~aux_formula,
      "age_slope_sigma_linear",   "(1+AGE|q|RING) + (1|YEAR)",   ~ 1 + AGE + TIME + (1|q|RING),
  )
  
  model_name <- model_spec$name
  model_path <- file.path(output_dir, paste0(response_var, "_", sex, "_fit_", model_name, ".rds"))

  if (file.exists(model_path)) {
    message(paste0("Skipping '", model_name, "': Model file already exists."))
    return(invisible(NULL))
  }

  message(paste0("\n--- Fitting model: ", model_name, " ---"))
  
  formula_str <- paste(response_var, "~", FIXED_EFFECTS_GAUSSIAN, "+", model_spec$mean_random)
  brm_formula <- bf(formula_str, sigma = model_spec$aux_formula, nl = FALSE)
  
  model_priors <- get_prior(brm_formula, data = data, family = gaussian())

  fit <- tryCatch({
    brm(
      formula = brm_formula, data = data, prior = model_priors, family = gaussian(),
      iter = 6000, warmup = 2000, chains = 4, seed = 123, init = 0,
      control = list(adapt_delta = 0.99, max_treedepth = 15)
    )
  }, error = function(e) { message(paste0("ERROR fitting '", model_name, "': ", e$message)); NULL })

  if (!is.null(fit)) {
    saveRDS(fit, model_path)
  }
}
```

## Data Preparation and Model Execution

The workflow is executed for both males and females. For each sex, the data is prepared, and then the run_mismatch_sensitivity function is called for the two alternative mismatch variables (X15MISMATCH and X25MISMATCH).

```{r}
#| label: execute-mismatch-sensitivity-male
#| eval: false

# --- Load and Prepare Data for Males ---
dat_male <- read.csv(here::here("..","data", "Lay_date_mismatches_250702.csv")) %>%
  subset(SEX == 0) %>%
  mutate(
    across(c(X15MISMATCH, X25MISMATCH, AGE, AFR, LONGEVITY), as.numeric),
    TIME = scale(YEAR, center = TRUE, scale = FALSE)[, 1],
    AGE = scale(AGE, center = TRUE, scale = FALSE)[, 1],
    AFR = scale(AFR, center = TRUE, scale = FALSE)[, 1],
    LONGEVITY = scale(LONGEVITY, center = TRUE, scale = FALSE)[, 1],
    YEAR = as.factor(YEAR),
    RING = as.factor(RING)
  )

# --- Run Sensitivity Models for Males ---
run_mismatch_sensitivity(data = dat_male, response_var = "X15MISMATCH", sex = "male")
run_mismatch_sensitivity(data = dat_male, response_var = "X25MISMATCH", sex = "male")
```
